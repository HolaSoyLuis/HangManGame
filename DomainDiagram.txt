@startuml

class HangedManGame

class Player
class Giver
class Guesser

class SecretWord
class Character
class MaxAttemps
class CountAttemps
' class AttempedWord
' class AttempedCharacters
SecretWord *-- Character

' SecretWord o-- AttempedWord

class SetSecretWord
class GuessAWord
class GuessACharacter

HangedManGame o--"2" Player

Guesser ..> GuessACharacter
