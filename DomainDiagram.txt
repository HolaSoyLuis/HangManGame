@startuml

object HangedManGame

object Player
object Giver
object Guesser

object SecretWord
object Character
object MaxAttemps
object CountAttemps
' object AttempedWord
object AttempedCharacters
' object CharAttemp
' object WordAttemp

' SecretWord o-- AttempedWord
SecretWord o-- AttempedCharacters
' SecretWord o-- CharAttemp
' SecretWord o-- WordAttemp

object SetSecretWord
object AttempAWord

Guesser ..> AttempACharacter
