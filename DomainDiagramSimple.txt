@startuml hangman_exercise

object HangManGame
object Player
object Giver
object Guesser
object setSecretWord
object setMaxAttemps
object attempAWord
object attempACharacter

HangManGame *--"2" Player
Player <|-- Giver
Player <|-- Guesser
Giver o-- setSecretWord
Giver o-- setMaxAttemps
Guesser o-- attempAWord
Guesser o-- attempACharacter

object Board
object countAttemps
object maxAttemps
object Turn
object turnActions
object printSecretWord
object checkIfHasWon
object checkIfHasLost

Board *-- countAttemps
Board *-- maxAttemps
Board o-- Turn
Turn o-- turnActions
turnActions <|-- checkIfHasWon
turnActions <|-- checkIfHasLost
turnActions <|-- printSecretWord

object SecretWord
object word
object attempedCharacters

SecretWord *-- word
SecretWord o-- attempedCharacters

HangManGame *-- Board
Board o-- SecretWord
Player --o Board

@enduml
